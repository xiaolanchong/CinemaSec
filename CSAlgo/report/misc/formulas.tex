\begin{equation}
{\bf{T}}^i = \left( {\begin{array}{*{20}c}
   {a_0^i} & {a_1^i} & {a_2^i}  \\
   {a_3^i} & {a_4^i} & {a_5^i}  \\
   0       & 0       & 1  \\
\end{array}} \right),
\end{equation}
%
где $i={\{l,c,r\}}$ есть индекс шаблон левого, центрального или правого кресла соответственно. %%@
Точка шаблона задаётся как однородный вектор
\[
{\bf p} = \left( \begin{array}{c}
x \\ y \\ 1 \\
\end{array} \right),
\]
%
а точка подвергнутая трансформации вычисляется по формуле
%
\begin{equation}
{\bf \tilde p} = {\bf T}{\bf p},
\end{equation}
%
обычно $a_0, a_4 \approx 1$, $a_1, a_2, a_3, a_5 \approx 0$.
%
\[
\begin{array}{lll}
\ipl & - & \mbox{вектор \emph{k}-той точки шаблона левого кресла}, \\
\ipc & - & \mbox{вектор \emph{k}-той точки шаблона центрального кресла}, \\
\ipr & - & \mbox{вектор \emph{k}-той точки шаблона правого кресла}.
\end{array}
\]
%
\begin{equation}
{\cal E}_{image} = - \left| {\nabla \left( {G*I} \right)} \right|^{q \, = \, 1,2} 
\end{equation}
%
\begin{equation}
\begin{array}{lll}
{\cal E}_{template} & = & \sum \limits_{k=0}^{N-1} { \left \| {
{\left( {\tpl - \left \langle \tpl \right \rangle } \right)} -
{\left( {\tpc - \left \langle \tpc \right \rangle } \right)}} \right \|^2 } + \\
\hfill & \hfill & \sum \limits_{k=0}^{N-1} { \left \| {
\left( {\tpr - \left \langle \tpr \right \rangle } \right) -
\left( {\tpc - \left \langle \tpc \right \rangle } \right)} \right \|^2 } + \\
\hfill & \hfill & \sum \limits_{k=0}^{N-1} {
\left \| {2 \, \tpc - \tpl - \tpr } \right \|^2 } \\ 
\end{array}
\end{equation}
%
\begin{equation}
\begin{array}{lll}
{\cal E}_{template} & = & \sum \limits_{k=0}^{N-1} { \left \| {
{\bf T}^l {\left( {\ipl - \left \langle \ipl \right \rangle } \right)} -
{\bf T}^c {\left( {\ipc - \left \langle \ipc \right \rangle } \right)}} \right \|^2 } + \\
\hfill & \hfill & \sum \limits_{k=0}^{N-1} { \left \| {
{\bf T}^r \left( {\ipr - \left \langle \ipr \right \rangle } \right) -
{\bf T}^c \left( {\ipc - \left \langle \ipc \right \rangle } \right)} \right \|^2 } + \\
\hfill & \hfill & \sum \limits_{k=0}^{N-1} {
\left \| {2 \, {\bf T}^c \ipc - {\bf T}^l \ipl - {\bf T}^r \ipr } \right \|^2 } \\ 
\end{array}
\end{equation}
%
\begin{equation}
{\cal E} = {\cal E}_{image} + {\cal E}_{template}
\end{equation}
